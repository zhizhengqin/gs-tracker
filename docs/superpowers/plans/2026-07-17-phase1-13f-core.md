# GS-Tracker Phase 1: 13F 核心闭环 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现 GS-Tracker 的 13F 核心闭环：自动抓取 Goldman Sachs 最新 13F-HR 报告，解析持仓数据，持久化到 SQLite，生成中文 AI 分析报告和 HTML 可视化报告，并通过 FastAPI Web 服务访问。

**Architecture:** 模块化流水线。`SEC13FFetcher` 抓取并解析 SEC EDGAR 13F XML；`Storage` 持久化到 SQLite；`QuarterComparator` 计算季度变化；`GSAnalyzer` 调用 Claude API 生成中文分析；`ReportGenerator` 渲染 HTML；`Web` 提供报告列表和单份报告访问；`Scheduler` 每周轮询新季度报告。

**Tech Stack:** Python 3.11, httpx, pandas, anthropic, jinja2, matplotlib, apscheduler, fastapi, pytest, SQLite.

## Global Constraints

- Python 3.11+，类型注解，PEP 8
- 异步 HTTP 使用 httpx + asyncio
- SEC EDGAR 请求必须带 `User-Agent`（含联系方式），否则 403
- API 密钥只从环境变量读取，禁止硬编码

```python
import xml.etree.ElementTree as ET

async def parse_13f_infotable(self, xml_url: str) -> pd.DataFrame:
    response = await self.client.get(xml_url)
    response.raise_for_status()
    root = ET.fromstring(response.text.encode("utf-8"))
    ns = {"ns": "http://www.sec.gov/edgar/document/thirteenf/informationtable"}
    rows = []
    for table in root.findall("ns:infoTable", ns):
        rows.append({
            "name_of_issuer": self._get_text(table, "ns:nameOfIssuer", ns),
            "title_of_class": self._get_text(table, "ns:titleOfClass", ns, "COM"),
            "cusip": self._get_text(table, "ns:cusip", ns),
            "value": float(self._get_text(table, "ns:value", ns, "0")) * 1000,
            "shares": int(self._get_text(table, "ns:shrsOrPrnAmt/ns:sshPrnamt", ns, "0")),
            "investment_discretion": self._get_text(table, "ns:investmentDiscretion", ns, ""),
            "voting_authority_sole": int(self._get_text(table, "ns:votingAuthority/ns:Sole", ns, "0")),
            "voting_authority_shared": int(self._get_text(table, "ns:votingAuthority/ns:Shared", ns, "0")),
            "voting_authority_none": int(self._get_text(table, "ns:votingAuthority/ns:None", ns, "0")),
        })
    return pd.DataFrame(rows)


def _get_text(self, element, path, ns, default="") -> str:
    found = element.find(path, ns)
    return found.text.strip() if found is not None and found.text else default
```

```python
# src/storage.py
import logging
import sqlite3
from contextlib import contextmanager
from pathlib import Path
from typing import List

from src.config import DATABASE_URL, PROJECT_ROOT

logger = logging.getLogger(__name__)


def db_path() -> Path:
    url = DATABASE_URL
    if url.startswith("sqlite:///"):
        relative = url.replace("sqlite:///", "")
        return (PROJECT_ROOT / relative).resolve()
    raise ValueError(f"Unsupported DATABASE_URL: {DATABASE_URL}")


@contextmanager
def get_connection():
    path = db_path()
    path.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(path)
    conn.row_factory = sqlite3.Row
    try:
        yield conn
    finally:
        conn.close()


def init_db() -> None:
    with get_connection() as conn:
        conn.executescript("""
        CREATE TABLE IF NOT EXISTS quarterly_reports (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            cik TEXT NOT NULL,
            quarter TEXT NOT NULL,
            filing_date TEXT,
            period_of_report TEXT,
            total_value REAL,
            num_holdings INTEGER,
            xml_url TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            UNIQUE(cik, quarter)
        );
        CREATE TABLE IF NOT EXISTS holdings (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            report_id INTEGER NOT NULL,
            cusip TEXT NOT NULL,
            name_of_issuer TEXT,
            title_of_class TEXT DEFAULT 'COM',
            value REAL,
            shares INTEGER,
            investment_discretion TEXT,
            voting_authority_sole INTEGER DEFAULT 0,
            voting_authority_shared INTEGER DEFAULT 0,
            voting_authority_none INTEGER DEFAULT 0,
            FOREIGN KEY (report_id) REFERENCES quarterly_reports(id)
        );
        CREATE INDEX IF NOT EXISTS idx_holdings_report ON holdings(report_id);
        CREATE INDEX IF NOT EXISTS idx_quarterly_cik_quarter ON quarterly_reports(cik, quarter);
        """)
        conn.commit()


def save_holdings(cik: str, quarter: str, holdings: List[dict]) -> int:
    with get_connection() as conn:
        cur = conn.execute(
            "INSERT OR REPLACE INTO quarterly_reports (cik, quarter, num_holdings, total_value) VALUES (?, ?, ?, ?)",
            (cik, quarter, len(holdings), sum(h.get("value", 0) for h in holdings)),
        )
        report_id = cur.lastrowid
        conn.execute("DELETE FROM holdings WHERE report_id = ?", (report_id,))
        for h in holdings:
            conn.execute(
                """
                INSERT INTO holdings (report_id, cusip, name_of_issuer, title_of_class, value, shares, investment_discretion,
                voting_authority_sole, voting_authority_shared, voting_authority_none)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
                """,
                (
                    report_id,
                    h["cusip"],
                    h.get("name_of_issuer", ""),
                    h.get("title_of_class", "COM"),
                    h.get("value", 0),
                    h.get("shares", 0),
                    h.get("investment_discretion", ""),
                    h.get("voting_authority_sole", 0),
                    h.get("voting_authority_shared", 0),
                    h.get("voting_authority_none", 0),
                ),
            )
        conn.commit()
        return report_id


def get_holdings(cik: str, quarter: str) -> List[dict]:
    with get_connection() as conn:
        row = conn.execute(
            "SELECT id FROM quarterly_reports WHERE cik = ? AND quarter = ?",
            (cik, quarter),
        ).fetchone()
        if not row:
            return []
        rows = conn.execute(
            "SELECT * FROM holdings WHERE report_id = ?",
            (row["id"],),
        ).fetchall()
        return [dict(r) for r in rows]
```

```python
# src/quarter_compare.py
from dataclasses import dataclass
from typing import List

import pandas as pd


@dataclass
class QuarterComparison:
    quarter_current: str
    quarter_previous: str
    new_positions: pd.DataFrame
    sold_positions: pd.DataFrame
    increased_positions: pd.DataFrame
    decreased_positions: pd.DataFrame
    concentration_change: float


class QuarterComparator:
    def __init__(self, value_threshold: float = 0.25) -> None:
        self.value_threshold = value_threshold

    def compare(
        self,
        current_df: pd.DataFrame,
        previous_df: pd.DataFrame,
        quarter_current: str,
        quarter_previous: str,
    ) -> QuarterComparison:
        cur = current_df.copy()
        prev = previous_df.copy()

        cur_cusips = set(cur["cusip"])
        prev_cusips = set(prev["cusip"])

        new_positions = cur[cur["cusip"].isin(cur_cusips - prev_cusips)].reset_index(drop=True)
        sold_positions = prev[prev["cusip"].isin(prev_cusips - cur_cusips)].reset_index(drop=True)

        merged = cur.merge(prev, on="cusip", how="inner", suffixes=("", "_prev"))
        merged["value_change_pct"] = (merged["value"] - merged["value_prev"]) / merged["value_prev"]

        increased = merged[merged["value_change_pct"] >= self.value_threshold].copy()
        decreased = merged[merged["value_change_pct"] <= -self.value_threshold].copy()

        hhi_current = self._compute_hhi(cur)
        hhi_previous = self._compute_hmi(prev)

        return QuarterComparison(
            quarter_current=quarter_current,
            quarter_previous=quarter_previous,
            new_positions=new_positions,
            sold_positions=sold_positions,
            increased_positions=increased,
            decreased_positions=decreased,
            concentration_change=hhi_current - hhi_previous,
        )

    def _compute_hhi(self, df: pd.DataFrame) -> float:
        total = df["value"].sum()
        if total == 0:
            return 0.0
        shares = (df["value"] / total) ** 2
        return float(shares.sum())
```

```python
# src/analyzer.py
import logging
from dataclasses import dataclass
from typing import List, Optional

import anthropic
import pandas as pd

from src.config import ANTHROPIC_API_KEY

logger = logging.getLogger(__name__)


@dataclass
class AnalysisResult:
    summary: str
    concentration_analysis: str
    top_holdings_analysis: str
    sector_preference: str
    trading_signals: str
    risk_warnings: str
    retail_insights: str
    key_tickers: List[str]
    sentiment: str
    confidence: float


class GSAnalyzer:
    def __init__(self, api_key: Optional[str] = None, model: str = "claude-sonnet-4-20250514"):
        self.client = anthropic.AsyncAnthropic(api_key=api_key or ANTHROPIC_API_KEY)
        self.model = model

    async def analyze_holdings(
        self,
        holdings_df: pd.DataFrame,
        previous_df: Optional[pd.DataFrame] = None,
    ) -> AnalysisResult:
        top10 = holdings_df.nlargest(10, "value")
        prompt = self._build_prompt(holdings_df, top10)
        response = await self.client.messages.create(
            model=self.model,
            max_tokens=4000,
            temperature=0.3,
            messages=[{"role": "user", "content": prompt}],
        )
        text = response.content[0].text
        return AnalysisResult(
            summary=text[:200],
            concentration_analysis=text,
            top_holdings_analysis=text,
            sector_preference=text,
            trading_signals=text,
            risk_warnings=text,
            retail_insights=text,
            key_tickers=top10["name_of_issuer"].tolist(),
            sentiment="neutral",
            confidence=0.5,
        )

    def _build_prompt(self, holdings_df: pd.DataFrame, top10: pd.DataFrame) -> str:
        total_value = holdings_df["value"].sum()
        return f"""你是一位顶级对冲基金分析师。请基于以下 Goldman Sachs 13F 持仓数据生成中文分析，控制在 800 字以内：

总持仓价值: ${total_value / 1e9:.2f}B
持仓标的数: {len(holdings_df)}
前10大持仓:
{top10[["name_of_issuer", "value"]].to_string(index=False)}

请分析：1) 持仓集中度 2) 重点标的 3) 行业偏好 4) 调仓信号 5) 风险提示 6) 对散户的启示。"""
```

```python
import logging
from pathlib import Path
from typing import List, Optional

import matplotlib
matplotlib.use("Agg")
import matplotlib.pyplot as plt
import pandas as pd
from jinja2 import Environment, FileSystemLoader, select_autoescape

from src.analyzer import AnalysisResult
from src.config import PROJECT_ROOT, REPORT_OUTPUT_DIR

logger = logging.getLogger(__name__)


class ReportGenerator:
    def __init__(self, template_dir: Optional[Path] = None):
        self.template_dir = template_dir or (PROJECT_ROOT / "templates")
        self.env = Environment(
            loader=FileSystemLoader(self.template_dir),
            autoescape=select_autoescape(["html", "xml"]),
        )

    def generate_report(
        self,
        quarter: str,
        holdings_df: pd.DataFrame,
        analysis: AnalysisResult,
        output_path: Optional[Path] = None,
    ) -> Path:
        output_path = output_path or (REPORT_OUTPUT_DIR / f"{quarter}.html")
        output_path.parent.mkdir(parents=True, exist_ok=True)
        template = self.env.get_template("report.html")
        html = template.render(
            quarter=quarter,
            holdings=holdings_df.to_dict("records"),
            analysis=analysis,
            generated_at="2026-01-01",
        )
        output_path.write_text(html, encoding="utf-8")
        return output_path
```

```python
import logging
from pathlib import Path
from typing import List

from fastapi import FastAPI, HTTPException
from fastapi.responses import HTMLResponse

from src.config import REPORT_OUTPUT_DIR

logger = logging.getLogger(__name__)

app = FastAPI(title="GS-Tracker", version="0.1.0")


@app.get("/", response_class=HTMLResponse)
async def list_reports() -> str:
    reports = sorted(REPORT_OUTPUT_DIR.glob("*.html"))
    items = [f'<li><a href="/reports/{r.name}">{r.stem}</a></li>' for r in reports]
    return f"<html><body><h1>GS-Tracker 报告列表</h1><ul>{''.join(items)}</ul></body></html>"


@app.get("/reports/{quarter}.html", response_class=HTMLResponse)
async def get_report(quarter: str) -> str:
    report_path = REPORT_OUTPUT_DIR / f"{quarter}.html"
    if not report_path.exists():
        raise HTTPException(status_code=404, detail="报告未找到")
    return report_path.read_text(encoding="utf-8")


@app.get("/api/reports")
async def api_reports() -> List[dict]:
    reports = sorted(REPORT_OUTPUT_DIR.glob("*.html"))
    return [
        {"quarter": r.stem, "title": f"高盛动向情报板 — {r.stem}", "path": f"/reports/{r.name}"}
        for r in reports
    ]


@app.get("/api/health")
async def health() -> dict:
    return {"status": "ok"}
```
